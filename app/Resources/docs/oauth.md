    http://scrumbag.dev/app_dev.php/api/products?access_token=MmI4YjFhN2MwOTU4Yjc1MmQyYzk1M2ZiNDRhMTAwMzIwMzRkYmE5MzIzNmI3ZDE1ZDFjYjExY2UyMzEyNzc4OQ
